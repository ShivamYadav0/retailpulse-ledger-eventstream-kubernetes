  name: ledger-service
spec:
  replicas: 2
  selector:
    matchLabels:
      app: ledger-service
  template:
    metadata:
      labels:
        app: ledger-service
    spec:
      containers:
        - name: ledger-service
          image: ledger-service:latest
          ports:
            - containerPort: 8080
          env:
            - name: REDIS_ADDR
              value: redis:6379
            - name: KAFKA_BROKERS
              value: kafka.vani-ledger.svc.cluster.local:9092
            - name: PG_CONN
              value: postgresql://ledger:ledgerpass@pgbouncer:6432/ledger
            - name: VOICE_REQUEST_TOPIC
              value: voice-requests
            - name: VOICE_RESPONSE_TOPIC
              value: voice-results
            - name: VOICE_POLL_INTERVAL_MS
              value: "1000"
